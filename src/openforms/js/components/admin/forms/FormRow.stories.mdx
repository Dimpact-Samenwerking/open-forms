# Form row

The form row acts as a container for one or more nested fields.

export const Template = args => (
  <FormRow {...args}>
    <Field errors={['Foo']} name="Field" label="Input field" helpText="Lorem ipsum">
      <TextInput />
    </Field>
  </FormRow>
);
